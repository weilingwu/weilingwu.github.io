三数之和：#

var threeSum = function(nums) {
    const len = nums.length;
    if(len < 3) return [];
    nums.sort((a, b) => a - b);
    const resSet = new Set();
    for(let i = 0; i < len - 2; i++) {
        if(nums[i] > 0) break;
        let l = i + 1, r = len - 1;
        while(l < r) {
            const sum = nums[i] + nums[l] + nums[r];
            if(sum < 0) { l++; continue };
            if(sum > 0) { r--; continue };
            resSet.add(`${nums[i]},${nums[l]},${nums[r]}`);
            l++;
            r--;
        }
    }
    return Array.from(resSet).map(i => i.split(","));
};

map的使用：

var majorityElement = function(nums) {
    if(nums.length < 3){
        return nums[0];
    }

    var map = new Map()
    var count = Math.floor(nums.length/2);
    for(let i = 0;i<nums.length;i++){
        if(map.get(nums[i])){
            map.set(nums[i], map.get(nums[i]) + 1)
            //修改value值
        }
        else{
            map.set(nums[i], 1)
        }
    }
    console.log(count)
    for (let key of map.keys()) {
        if(map.get(key)>count){
            //获取value的正确写法
            return key;
        }
    }
};

子数组最大和：

const maxSubArray = nums => {
    // 数组长度，dp初始化
    const [len, dp] = [nums.length, [nums[0]]];
    // 最大值初始化为dp[0]
    let max = dp[0];
    for (let i = 1; i < len; i++) {
        dp[i] = Math.max(dp[i - 1] + nums[i], nums[i]);
        // 更新最大值
        max = Math.max(max, dp[i]);
    }
    return max;
};

环形数组：

两种情况：一种是最大和子数组在数组内；一种是最大子数组在两边，这个时候除开最大子数组，中间的留下的就是最小子数组。

var total = 0, maxSum = nums[0], curMax = 0, minSum = nums[0], curMin = 0;
for (let a of nums) {
    curMax = Math.max(curMax + a, a);
    maxSum = Math.max(maxSum, curMax);
    curMin = Math.min(curMin + a, a);
    minSum = Math.min(minSum, curMin);
    total += a;
}
return maxSum > 0 ? Math.max(maxSum, total - minSum) : maxSum;
